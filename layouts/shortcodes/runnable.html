package main

import (
	"context"
	"flag"
	"fmt"
	"log"

	"github.com/dgraph-io/dgraph/client"
	"github.com/dgraph-io/dgraph/protos/graphp"
	"github.com/gogo/protobuf/proto"
	"google.golang.org/grpc"
)

var (
	dgraph = flag.String("d", "127.0.0.1:8080", "Dgraph server address")
)

func main() {
	conn, err := grpc.Dial("127.0.0.1:8080", grpc.WithInsecure())
	if err != nil {
		log.Fatal(err)
	}
	defer conn.Close()

	dgraphClient := graphp.NewDgraphClient(conn)
	req := client.Req{}
	req.SetQuery(`{{ $code }}`)

	resp, err := dgraphClient.Run(context.Background(), req.Request())
	if err != nil {
		log.Fatalf("Error in getting response from server, %s", err)
	}
	fmt.Printf("Response %+v\n", proto.MarshalTextString(resp))
}
package main

import (
        "bytes"
        "flag"
        "log"
        "net/http"
)

var (
        baseURL = flag.String("d", "http://127.0.0.1:8080/", "Dgraph server address")
)

func main() {
        client := &http.Client{}
        body := bytes.NewBufferString(`{{ $code }}`)
        req, err := http.NewRequest("POST", *baseURL, body)
        if err != nil {
                log.Fatal(err)
                return
        }

        resp, err := client.Do(req)
        if err != nil {
                log.Fatal(err)
                return
        }
        defer resp.Body.Close()

        log.Printf("Response %+v\n", resp)
}
